Overview

examp is a simple R package including two functions. Bobble is a simple function developed to sort the given array of values. fac is a function developed to calculate the factorial. fcount , fEcount , fE2count ,are three different kernel functions. ## Benchmarking Bobble and fac

The source R code for Bobble is as follows:

Bobble<-function(L,n){
 library(Rcpp)
  cppFunction('NumericMatrix Bobble(NumericMatrix x,int m){
  NumericMatrix mat(m, 1);
  NumericMatrix Index(m,1);
  for (int k=0;k<m;k++){
    mat(k,0)=x(k,0);
    Index(k,0)=k+1;
  }
  int i,j,t;
  double l;
  for(i=0;i<(m-1);i++){
     for(j=0;j<(m-1-i);j++){
        if(mat(j,0)>mat((j+1),0)){
        l=mat(j,0);mat(j,0)=mat((j+1),0);mat((j+1),0)=l;
        t=Index(j,0);Index(j,0)=Index((j+1),0);Index((j+1),0)=t;
        }}}
  return(Index);
}')
LL<-numeric(n)
Ind<-Bobble(L,n)
return(Ind)
}

fac<-function(n){
library(Rcpp)
cppFunction('int fac(int m){
  int P;
  if((m==0)||(m==1)){
    P=1;
  }
  else {P=m*fac(m-1);}
  
  return(P);
}')

y<-fac(n)

return(y)
}

fcount<-function(a1,a2,b1,b2,ha,hb){#Multiplicative kernel
  s<-0
  bar1<-(b1-a1)/ha
  bar2<-(b2-a2)/hb
  bar<-((-1<=bar1)&(bar1<1)&(-1<=bar2)&(bar2<1))
  if (bar){s<-1}
  return(s/(ha*hb))
}

fEcount<-function(a1,a2,b1,b2,ha,hb){#The multivariate Epanechnikov (spherical)
  s<-0
  bar1<-(b1-a1)/ha
  bar2<-(b2-a2)/hb
  bar3<-c(bar1,bar2)
  bar<-t(bar3)%*%(bar3)
  if(bar<=1){
    s<-(1-bar)
  }
  return(s/(ha*hb))
}

fE2count<-function(a1,a2,b1,b2,ha,hb){#The multivariate Epanechnikov (multiplicative)
  s<-0
  bar1<-(b1-a1)/ha
  bar2<-(b2-a2)/hb
  bar<-((abs(bar1)<=1)&(abs(bar2)<=1))
  if (bar){
    s<-(3/4)^2*(1-(bar1)^2)*(1-(bar2)^2)
  }
  return(s/(ha*hb))
}
